<script>

const bullyWords = [
"stupid","idiot","loser","ugly","hate","kill","worthless",
"moron","dumb","trash","pathetic","fool","shut","bitch",
"asshole","jerk","retard","scum","garbage","rubbish",
"bastard","fuck","shit","crybaby"
];

let recognition;
let active = false;

/* ---------------- VOICE DETECTION ---------------- */

function startMic(){

const SpeechRecognition = window.SpeechRecognition || window.webkitSpeechRecognition;

recognition = new SpeechRecognition();

recognition.lang="en-US";
recognition.continuous=true;
recognition.interimResults=true;

recognition.start();
active=true;

recognition.onresult=function(event){

let text="";

for(let i=0;i<event.results.length;i++){

text+=event.results[i][0].transcript+" ";

}

text=normalize(text);

document.getElementById("speechOutput").innerText=text;

detectBullying(text);

};

}

function stopMic(){

if(recognition && active){
recognition.stop();
active=false;
}

}

/* ---------------- TEXT INPUT ---------------- */

function analyzeText(){

let text=document.getElementById("textInput").value;

text=normalize(text);

document.getElementById("speechOutput").innerText=text;

detectBullying(text);

}

/* ---------------- DATASET FILE ---------------- */

function analyzeDataset(){

let file=document.getElementById("datasetFile").files[0];

if(!file){

alert("Please upload dataset");

return;

}

let reader=new FileReader();

reader.onload=function(e){

let text=e.target.result;

text=normalize(text);

document.getElementById("speechOutput").innerText=text.substring(0,200)+"...";

detectBullying(text);

};

reader.readAsText(file);

}

/* ---------------- TEXT CLEANING ---------------- */

function normalize(text){

text=text.toLowerCase();

text=text.replace(/[^\w\s]/g,"");

return text;

}

/* ---------------- BULLY DETECTION ---------------- */

function detectBullying(sentence){

let tokens=sentence.split(/\s+/);

let detected=[];

tokens.forEach(word=>{

bullyWords.forEach(b=>{

if(word===b || word.includes(b)){

detected.push(b);

}

});

});

displayResults(detected);

}

/* ---------------- OUTPUT DISPLAY ---------------- */

function displayResults(found){

let keywordBox=document.getElementById("keywords");

keywordBox.innerHTML="";

let unique=[...new Set(found)];

unique.forEach(word=>{

let span=document.createElement("span");

span.className="keyword";

span.innerText=word;

keywordBox.appendChild(span);

});

document.getElementById("count").innerText=
found.length+" bullying words detected";

let risk=Math.min(found.length*20,100);

document.getElementById("riskScore").innerText=risk+"%";

}
